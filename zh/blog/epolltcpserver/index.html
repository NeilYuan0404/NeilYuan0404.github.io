用epoll实现一个TCP服务器

2026年1月18日

一、了解epoll#

关于I/O多路复用（epoll） epoll的核心优势在于其高效的数据结构：使用红黑树来管理全部需要监控的描述符（简称为整fd集）、双向链表来管理已就绪的描述符（就绪fd集），并且在事件就绪的时候由内核直接通知用户空间，避免不必要的内存复制。支持水平触发LT和边沿触发ET，值得注意的是边沿触发ET：仅在fd状态发生变化的时候通知一次，避免了事件的重复触发，需要配合I/O非阻塞处理使用。

与epoll有关的函数：

 1//创建epoll实例，得到epfd
 2int epoll_create(int size)
 3
 4//控制事件（操作红黑树）
 5//参数op：EPOLL_CTL_ADD/MOD/DEL
 6int epoll_ctl(int epfd, int op, int fd, struct epoll_event *event)
 7
 8//等待事件（返回就绪集链表）
 9int epoll_wait(int epfd, struct epoll_event *events,
10               int maxevents, int timeout)

2.2 事件#

1// epoll_event结构体
2struct epoll_event {
3    uint32_t events;    // 事件类型（EPOLLIN等，见下）
4    epoll_data_t data;  // 附加数据（用户数据）
5};

在网络编程中，事件主要是：

  1. 可读事件（EPOLLIN）：有数据可以读了
  2. 可写事件（EPOLLOUT）：可以发送数据了
  3. 错误事件（EPOLLERR）：出错了
  4. 挂起事件（EPOLLHUP）：连接断开了

二、实现#

2.1 多端口监听#

使用epoll_create函数来创建一个epoll实例，其参数大于0即可，我们可取1，这个实例用epfd去描述。

然后使用socket函数并且指定参数SOCK_STREAM，这样会自动选择建立TCP连接。当我们只需要去监听一个端口的时候，在addr.sin_port当中指定我们想要的端口即可。

这里为了提升TCP服务器的性能，我们采用for循环创建多个端口（port+i），并且将这些端口与sockfd绑定，使用listen监听，再将这些端口的事件都会被加入epoll实例中。

 1if (argc < 2) {
 2		printf("Param Error\n");
 3		return -1;
 4	}
 5	
 6	int port = atoi(argv[1]); // start 
 7	int sockfds[MAX_PORT] = {0}; // listen fd
 8	int epfd = epoll_create(1);  
 9
10	int i = 0;
11	for (i = 0;i < MAX_PORT;i ++) {
12
13		int sockfd = socket(AF_INET, SOCK_STREAM, 0);
14
15		struct sockaddr_in addr;
16		memset(&addr, 0, sizeof(struct sockaddr_in));
17		addr.sin_family = AF_INET;
18		addr.sin_port = htons(port+i); // 8888 8889 8890 8891 .... 8987
19		addr.sin_addr.s_addr = INADDR_ANY; 
20
21		if (bind(sockfd, (struct sockaddr*)&addr, sizeof(struct sockaddr_in)) < 0) {
22			perror("bind");
23			return 2;
24		}
25
26		if (listen(sockfd, 5) < 0) {
27			perror("listen");
28			return 3;
29		}
30		printf("tcp server listen on port : %d\n", port + i);
31
32		struct epoll_event ev;
33		ev.events = EPOLLIN; 
34		ev.data.fd = sockfd;
35		epoll_ctl(epfd, EPOLL_CTL_ADD, sockfd, &ev);
36
37		sockfds[i] = sockfd;
38	}

2.2 主事件循环#

先声明一个事件数组（总集），进主事件循环。

我们使用epoll_wait函数，当事件就绪的时候（EPOLLIN，EPOLLOUT）需要处理这些事件，指定timeout = 5，避免不必要的系统资源占用。

我们判断是否遇到监听套接字listenfd，如果是监听套接字，则意味着这是一个新的连接，使用accept函数接收这个新链接，转化为客户端套接字。注意，这里处理监听套接字是使用非阻塞的方式（fcntl函数就是控制为非阻塞方式），因为我们采用epoll边沿触发（EPOLLET），必须一次性读完全部数据。而后使用epoll_ctl函数将事件加入。

当遇到客户端套接字clientfd的时候，说明有客户端数据需要发送，调用recv函数接收即可。注意这接收应该采用水平触发的方式，一直有数据就一直触发，一直接收。完毕后使用epoll_ctl删去。

 1struct epoll_event events[EPOLL_SIZE] = {0};
 2
 3	while (1) {
 4
 5		int nready = epoll_wait(epfd, events, EPOLL_SIZE, 5); // -1, 0, 5
 6		if (nready == -1) continue;
 7
 8		int i = 0;
 9		for (i = 0;i < nready;i ++) {
10
11			int sockfd = islistenfd(events[i].data.fd, sockfds);
12			if (sockfd) { // listen 2
13
14				struct sockaddr_in client_addr;
15				memset(&client_addr, 0, sizeof(struct sockaddr_in));
16				socklen_t client_len = sizeof(client_addr);
17
18				int clientfd = accept(sockfd, (struct sockaddr*)&client_addr, &client_len);
19
20				fcntl(clientfd, F_SETFL, O_NONBLOCK);
21
22				int reuse = 1;
23				setsockopt(clientfd, SOL_SOCKET, SO_REUSEADDR, (char *)&reuse, sizeof(reuse));
24
25				struct epoll_event ev;
26				ev.events = EPOLLIN | EPOLLET; 
27				ev.data.fd = clientfd;
28				epoll_ctl(epfd, EPOLL_CTL_ADD, clientfd, &ev);
29
30			} else {
31
32				int clientfd = events[i].data.fd;
33				
34				char buffer[BUFFER_LENGTH] = {0};
35				int len = recv(clientfd, buffer, BUFFER_LENGTH, 0);
36				if (len < 0) {
37					close(clientfd);
38
39					struct epoll_event ev;
40					ev.events = EPOLLIN; 
41					ev.data.fd = clientfd;
42					epoll_ctl(epfd, EPOLL_CTL_DEL, clientfd, &ev);
43					
44				} else if (len == 0) { // disconnect
45					close(clientfd);
46
47					struct epoll_event ev;
48					ev.events = EPOLLIN; 
49					ev.data.fd = clientfd;
50					epoll_ctl(epfd, EPOLL_CTL_DEL, clientfd, &ev);
51					
52				} else {
53					printf("Recv: %s, %d byte(s), clientfd: %d\n", buffer, len, clientfd);
54				}
55				
56				
57			}
58
59		}
60
61	}
62	return 0;
63}
