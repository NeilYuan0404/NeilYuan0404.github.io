一文带你搞清楚什么是线程池

2025年11月21日

一、底层数据结构#

线程池是一个服务器的基础组件，由任务队列（等待处理的任务）、工作队列（实际上的线程池）、管理组件所组成。

1.1 任务队列和工作队列#

定义两个双向链表来分别实现任务队列和工作队列。任务队列当中需要定义一个指针函数，用于回调执行任务的函数入口。工作队列当中定义了一个int terminate，作用是当线程中的工作已经完成，将其置位以打破循环退出。

 1//任务队列的结构体
 2 struct nTask {
 3  void (*task_func) (struct nTask* task);
 4  void *user_data;
 5
 6  struct nTask *prev;
 7  struct nTask *next;
 8};
 9//执行队列的结构体
10struct nWorker {
11  pthread_t threadid;
12
13  int terminate;
14  struct nManager *manager;
15  
16  struct nWorker *prev;
17  struct nWorker *next;
18};

1.2 管理组件#

包含了任务队列和工作队列。还需要定义一个条件变量cond（类型是pthread_cond_t），作用是为了让工作线程阻塞等待，配合一把互斥锁用于同步。

1typedef struct nManager {
2  struct nTask *tasks;
3  struct nWorker *workers;
4
5  pthread_mutex_t mutex;
6  pthread_cond_t cond; //条件变量
7} ThreadPool;

API实现#

线程池创建：int nThreadPoolCreate(ThreadPool *pool,int numWorkers) 回调：static void *nThreadPoolCallBack(void *arg)每个任务队列中的任务需要一个回调函数来“消费”它 任务推送：int nThreadPoolPushTask(ThreadPool *pool, struct nTask *tasks)当线程池拥有可用的线程的时候，将任务推送到工作队列当中执行 销毁线程池：int nThreadPoolDestory(ThreadPool *pool, int nWorker)

2.1 线程池创建#

返回值：成功时返回0，错误返回非0值 参数：线程池实例pool，最大的工作线程数目numWorks

需要进行手动内存管理，使用memset将threadpool结构体所需要的内存大小初始化为空内存，这是为了避免出现脏数据。

为实例pool创建数量为numWorks的工作线程worker，通过for循环调用malloc函数n分配内存、pthread_create创建工作线程，指定需要的回调函数。

最后，将创建好的工作线程插入pool的链表中，即创建完毕。

 1int nThreadPoolCreate(ThreadPool *pool,int numWorkers) {
 2  if (pool == NULL) return -1;
 3  if (numWorkers < 1) numWorkers = 1;
 4  memset(pool, 0, sizeof(ThreadPool));
 5
 6  pthread_cond_t blank_cond = PTHREAD_COND_INITIALIZER;
 7  memcpy(&pool->cond, &blank_cond, sizeof(pthread_cond_t));
 8
 9  pthread_mutex_init(&pool->mutex, NULL);
10
11  int i = 0;
12  for (i = 0;i < numWorkers;i++) {
13    struct nWorker *worker =(struct nWorker*)malloc(sizeof(struct nWorker));
14    if (worker == NULL) {
15      perror("malloc");
16      return -2;
17    }
18    memset(worker, 0, sizeof(struct nWorker));
19    worker->manager = pool;
20
21    int ret = pthread_create(&worker->threadid, NULL, nThreadPoolCallBack, worker);
22    if (ret) {
23      perror("pthread_create");
24      free(worker);
25      return -3;
26    }
27    LIST_INSERT(worker, pool->workers);
28  }
29  printf("call_back\n");
30  return 0;
31}

2.2 回调函数#

while（1）循环判断任务队列当中是否有任务，如果没有任务就阻塞等待（之前的pthread_cond_wait条件等待），有任务就执行任务。这个过程需要加锁保护，因为涉及到双向链表的操作

 1static void *nThreadPoolCallBack(void *arg) {
 2  struct nWorker *worker = (struct nWorker*)arg;
 3
 4  while (1) {
 5    pthread_mutex_lock(&worker->manager->mutex);
 6    while (worker->manager->tasks == NULL) {
 7      if (worker->terminate) break;
 8      pthread_cond_wait(&worker->manager->cond, &worker->manager->mutex);
 9    }
10    //避免死锁
11    if (worker->terminate) {
12      pthread_mutex_unlock(&worker->manager->mutex);
13      break;
14    }
15
16    struct nTask *task = worker->manager->tasks;
17    LIST_REMOVE(task, worker->manager->tasks);
18
19    pthread_mutex_unlock(&worker->manager->mutex);
20
21    task->task_func(task);
22  }
23
24  free(worker);
25
26  return NULL;
27}

2.3 任务推送#

推送任务的关键是依赖条件变量来唤醒线程pthread_cond_signal()，并且将任务添加到链表里，涉及底层链表的操作时不要忘记加锁保护。

 1int nThreadPoolPushTask(ThreadPool *pool, struct nTask *tasks) {
 2  pthread_mutex_lock(&pool->mutex);
 3
 4  LIST_INSERT(tasks, pool->tasks);
 5
 6  pthread_cond_signal(&pool->cond);//唤醒一个等待这个条件的线程
 7  
 8  pthread_mutex_unlock(&pool->mutex);
 9
10  return 0;
11}

2.4 线程池删除#

通过之前的代码我们已经知道，将所有work的terminate全部置位，则这些工作线程会自己退出。现在我们置位后使用pthread_cond_broadcast函数将所有阻塞在条件等待的线程全部唤醒，这些线程就会关闭并且从链表中删去，相比于直接将pool实例中的工作队列和任务队列全部置为NULL，这样做才能避免内存泄漏的危险。

 1int nThreadPoolDestory(ThreadPool *pool, int nWorker) {
 2  struct nWorker *worker = NULL;
 3  for (worker = pool->workers; worker != NULL; worker = worker->next) {
 4    worker->terminate = 1;
 5  }
 6  //这把锁和条件等待时候的锁是同一把
 7  pthread_mutex_lock(&pool->mutex);
 8
 9  pthread_cond_broadcast(&pool->cond);//唤醒所有等待这个条件的线程
10  
11  pthread_mutex_unlock(&pool->mutex);
12
13  pool->workers = NULL;
14  pool->tasks = NULL;
15
16  return 0;
17}
